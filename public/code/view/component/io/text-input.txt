<template>
  <AppInputText
    v-model="input.state"
    name="text-demo"
    v-bind="input"
  />
</template>

<script lang="ts" setup>
import { reactive } from 'vue';
import AppInputText from '@core/form/AppInputText.vue';
import { createInputState } from '@/composables/useInput';
import type { TextInputModel } from '@types';

function validateTextInput(value: string) {
  if (value.length === 0) {
    return 'String is empty';
  }
  return null;
}

const input = reactive<TextInputModel>({
  state: createInputState({ value: '' }),
  validate: validateTextInput,
});

</script>
